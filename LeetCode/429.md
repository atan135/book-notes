## 429. N-ary Tree Level Order Traversal  

**题目：**

![](http://p9zl5r4hu.bkt.clouddn.com/2018-10-01_20leet_429.png)

**解答：**

这个题很简单，就是一个简单的层状遍历，但是通过优化代码可以减少代码的运行时间。

**代码：**

```cpp
// Definition for a Node.
class Node {
public:
    int val = NULL;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
//方案一：使用一个vector保存每层数据，测试耗时80ms，程序需要大量的拷贝和复制vector
class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
        vector<vector<int>> result;
        if(root == NULL)    return result;
        vector<Node*> vecLevel;
        vecLevel.push_back(root);
        while(!vecLevel.empty()){
            vector<int> vecVal;
            vector<Node*> nextLevel;
            for(int i = 0;i < vecLevel.size(); ++i){
                vecVal.push_back(vecLevel[i]->val);
                if(vecLevel[i]->children.size() > 0){
                    for(int j = 0;j < vecLevel[i]->children.size(); ++j)
                        nextLevel.push_back(vecLevel[i]->children[j]);
                }
            }
            vecLevel = nextLevel;
            result.push_back(vecVal);
        }
        return result;
    }
};
//方案二：使用queue来保持先进先出的一个节点队列，使用levelCnt来标记每层的数量，代码简洁度增加，且运行速度更快，耗时52ms
class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
        vector<vector<int>> result;
        if(root == NULL)    return result;
        queue<Node*> q;
        q.push(root);
        int levelCnt = 1;
        while(!q.empty()){
            vector<int> vecVal;
            int nextLevelCnt = 0;
            while(levelCnt > 0){
                --levelCnt;
                Node* node = q.front();q.pop();
                vecVal.push_back(node->val);
                for(auto child : node->children){
                    q.push(child);
                    ++nextLevelCnt;
                }
            }
            levelCnt = nextLevelCnt;
            result.push_back(vecVal);
        }
        return result;
    }
};
//方案三：进一步优化，减少中间vector新建次数，每次用完clear掉即可，耗时48ms
class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
        vector<vector<int>> result;
        if(root == NULL)    return result;
        queue<Node*> q;
        q.push(root);
        int levelCnt = 1;
        vector<int> vecVal;
        while(levelCnt > 0){
            int nextLevelCnt = 0;
            while(levelCnt > 0){
                --levelCnt;
                Node* node = q.front();q.pop();
                vecVal.push_back(node->val);
                for(auto child : node->children){
                    q.push(child);
                    ++nextLevelCnt;
                }
            }
            levelCnt = nextLevelCnt;
            result.push_back(vecVal);
            vecVal.clear();
        }
        return result;
    }
};
```
