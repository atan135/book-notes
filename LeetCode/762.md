## 762. Prime Number of Set Bits in Binary Representation

**题目：**

![](http://p9zl5r4hu.bkt.clouddn.com/2018-09-30leet_762.png)

**解答：**

这道题就是计算数字二进制表示中1的个数，主要考察的是怎样优化算法效率。

**代码：**

```cpp
class Solution {
public:
    int countPrime(int n){
        int cnt = 0;
        if( n & 1)  cnt++;
        if( n & 2)  cnt++;
        if( n & 4)  cnt++;
        if( n & 8)  cnt++;
        if( n & 16)  cnt++;
        if( n & 32)  cnt++;
        if( n & 64)  cnt++;
        if( n & 128)  cnt++;
        if( n & 256)  cnt++;
        if( n & 512)  cnt++;
        if( n & 1024)  cnt++;
        if( n & 2048)  cnt++;
        if( n & 4096)  cnt++;
        if( n & 8192)  cnt++;
        if( n & 16384)  cnt++;
        if( n & 32768)  cnt++;
        if( n & 65536)  cnt++;
        if( n & 131072)  cnt++;
        if( n & 262144)  cnt++;
        if( n & 524288)  cnt++;
        if(cnt == 2 || cnt == 3 || cnt == 5 || cnt == 7 || cnt == 11 || cnt == 13 || cnt == 17 || cnt == 19 || cnt == 23 || cnt == 29)
            return true;
        return false;
    }
    int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for(int i = L; i <= R; ++i){
            if(countPrime(i))
                ++cnt;
        }
        return cnt;
    }
};
```

这是一个可以保证效率的笨方法，然后可以提升计算的速度的方法如下：

```cpp
class Solution {
public:
    int countPrime(int n){
        int cnt = 0;
        while (n > 0)
        {
            cnt++;
            n = n & (n-1);
        }
        if(cnt == 2 || cnt == 3 || cnt == 5 || cnt == 7 || cnt == 11 || cnt == 13 || cnt == 17 || cnt == 19 || cnt == 23 || cnt == 29)
            return true;
        return false;
    }
    int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for(int i = L; i <= R; ++i){
            if(countPrime(i))
                ++cnt;
        }
        return cnt;
    }
};
```
